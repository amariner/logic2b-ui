## Installation

```bash
npx shadcn@latest add https://ui.logic2b.dev/r/aspect-ratio.json
```

## Usage

```tsx
import { AspectRatio } from "@/components/ui/aspect-ratio"

<AspectRatio ratio={16 / 9}>
  <img src="..." alt="Image" className="rounded-md object-cover" />
</AspectRatio>
```
