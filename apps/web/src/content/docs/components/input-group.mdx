## Usage

```tsx
import {
  InputGroup,
  InputGroupAddon,
  InputGroupInput,
  InputGroupText,
} from "@/components/ui/input-group"

<InputGroup>
  <InputGroupAddon>
    <SearchIcon />
  </InputGroupAddon>
  <InputGroupInput placeholder="Search..." />
</InputGroup>
```

## API

| Prop | Type | Default |
| --- | --- | --- |
| `align` (InputGroupAddon) | `inline-start \| inline-end \| block-start \| block-end` | `inline-start` |
| `size` (InputGroupButton) | `xs \| sm \| icon-xs \| icon-sm` | `xs` |
| `variant` (InputGroupButton) | `default \| secondary \| outline \| ghost \| link \| destructive` | `ghost` |
